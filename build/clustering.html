# this file.
LiCl <- read.delim("../data/LiCl.txt")

# The NAME column contains unique gene identifiers, so we'll use these as the
# row labels in our data.frame and remove the NAME column from the table.
rownames(LiCl) <- LiCl$NAME
LiCl$NAME <- NULL
# Convert the gene expression data to a 2D matrix; omit genes with missing
# values.  We'll trim off the rownames to make the dendograms readable.
m.LiCl <- as.matrix( na.omit(LiCl) )
rownames(m.LiCl) <- NULL
# Distance matrix for gene expression vectors (these are log2(Exp/Ref) values)
d <- dist(m.LiCl, method = "euclidean")
# Clustering on the distance matrix
fit <- hclust(d, method = "ward")
print(fit)

# Visualize the cluster with plot
plot(fit)


# Heatmap of the matrix
heatmap(m.LiCl)


# gplots provides the colorpanel function, making it easy to generate color
# gradients install.packages("gplots") # you made need to install this package
library(gplots)
# Use heatmap with a custom color gradient
heatmap( m.LiCl
       , col  = colorpanel(100, "royalblue3", "black", "yellow")
       )


# Suppress clustering on the column (experiment) axis
heatmap( m.LiCl
       , col  = colorpanel(100, "royalblue3", "black", "yellow")
       , Colv = NA
       )


# Specify color saturation points (all log2 values =>2.5 or =<-2.5 are drawn as
# the same color)
heatmap( m.LiCl
       , col  = colorpanel(100, "royalblue3", "black", "yellow")
       , Colv = NA
       , zlim = c(-2.5, 2.5)
       )


# Use RowSideColors to highlight genes with the "protein biosynthesis" GO
# process annotation
heatmap( m.LiCl
       , col  = colorpanel(100,"royalblue3","black","yellow")
       , Colv = NA
       , RowSideColors = ifelse( grepl("protein biosynthesis"
                               , rownames(LiCl))
                               , "darkgrey"
                               , "white"
                               )
       , labRow = c(""), labCol=c("")
       )


# Tree trimming a hierarchy
groups <- cutree(fit, k = 3)
plot(fit)
# Highlight k groups
rect.hclust(fit, k = 5, border = "red")
